#include <WiFi.h>
#include <HTTPClient.h>
#include <Wire.h>
#include "MAX30105.h"
#include <Adafruit_Sensor.h>
#include <Adafruit_BME680.h>

// ===== إعدادات الواي فاي والسيرفر =====
const char* ssid = "iphone......";
const char* password = "91929132";

// حط هنا IP اللابتوب أو السيرفر
const char* serverName = "http://172.20.10.2:3000/sensor-data";

// نفس API_KEY الموجود في ملف .env
const char* apiKey = "123456";

// ===== تعريف الحساسات =====
MAX30105 heartRate;
Adafruit_BME680 bme;

// ===== متغيرات القراءة =====
float temperature = 0.0;
float humidity = 0.0;
float gasResistance = 0.0;
long irValue = 0;

void connectToWiFi() {
  Serial.print("جاري الاتصال بالواي فاي");
  WiFi.begin(ssid, password);

  int retryCount = 0;
  while (WiFi.status() != WL_CONNECTED && retryCount < 20) {
    delay(1000);
    Serial.print(".");
    retryCount++;
  }

  if (WiFi.status() == WL_CONNECTED) {
    Serial.println();
    Serial.println("تم الاتصال بالواي فاي بنجاح.");
    Serial.print("IP Address: ");
    Serial.println(WiFi.localIP());
  } else {
    Serial.println();
    Serial.println("فشل الاتصال بالواي فاي.");
  }
}

void sendDataToServer(float temp, float hum, float gas, long ir) {
  if (WiFi.status() == WL_CONNECTED) {
    HTTPClient http;

    http.begin(serverName);
    http.addHeader("Content-Type", "application/json");

    String jsonData = "{";
    jsonData += "\"device_id\":\"esp32_01\",";
    jsonData += "\"temperature\":" + String(temp, 2) + ",";
    jsonData += "\"humidity\":" + String(hum, 2) + ",";
    jsonData += "\"gas_resistance\":" + String(gas, 2) + ",";
    jsonData += "\"ir_value\":" + String(ir) + ",";
    jsonData += "\"api_key\":\"" + String(apiKey) + "\"";
    jsonData += "}";

    Serial.println("البيانات المرسلة للسيرفر:");
    Serial.println(jsonData);

    int httpResponseCode = http.POST(jsonData);

    Serial.print("Response Code: ");
    Serial.println(httpResponseCode);

    String response = http.getString();
    Serial.print("Server Response: ");
    Serial.println(response);

    http.end();
  } else {
    Serial.println("الواي فاي غير متصل، لا يمكن إرسال البيانات.");
  }
}

void setup() {
  Serial.begin(115200);
  while (!Serial);

  Serial.println(F("--- فحص الحساسات وربط الشبكة ---"));

  // تشغيل I2C
  Wire.begin();

  // الاتصال بالواي فاي
  connectToWiFi();

  // 1. إعداد الحساس البنفسجي BME680
  if (!bme.begin()) {
    Serial.println(F("خطأ: لم يتم العثور على BME680!"));
  } else {
    bme.setTemperatureOversampling(BME680_OS_8X);
    bme.setHumidityOversampling(BME680_OS_2X);
    bme.setPressureOversampling(BME680_OS_4X);
    bme.setIIRFilterSize(BME680_FILTER_SIZE_3);
    bme.setGasHeater(320, 150);
    Serial.println(F("تم تفعيل BME680 بنجاح."));
  }

  // 2. إعداد الحساس الأسود MAX30105
  if (!heartRate.begin(Wire, I2C_SPEED_STANDARD)) {
    Serial.println(F("خطأ: لم يتم العثور على حساس ضربات القلب!"));
  } else {
    heartRate.setup();
    Serial.println(F("تم تفعيل حساس ضربات القلب بنجاح."));
  }
}

void loop() {
  // إذا انقطع الواي فاي، حاول ترجع تربطه
  if (WiFi.status() != WL_CONNECTED) {
    Serial.println("انقطع الواي فاي، جاري إعادة الاتصال...");
    connectToWiFi();
  }

  // ===== قراءة بيانات BME680 =====
  if (bme.performReading()) {
    temperature = bme.temperature;
    humidity = bme.humidity;
    gasResistance = bme.gas_resistance / 1000.0;

    Serial.print(F("الحرارة: "));
    Serial.print(temperature);
    Serial.print(F(" C | "));

    Serial.print(F("الرطوبة: "));
    Serial.print(humidity);
    Serial.print(F(" % | "));

    Serial.print(F("الغاز: "));
    Serial.print(gasResistance);
    Serial.println(F(" KOhms"));
  } else {
    Serial.println(F("فشل في قراءة BME680"));
  }

  // ===== قراءة حساس MAX30105 =====
  irValue = heartRate.getIR();

  if (irValue < 50000) {
    Serial.println(F("وضع الإصبع: لا يوجد"));
  } else {
    Serial.print(F("إشارة النبض (IR): "));
    Serial.println(irValue);
  }

  // ===== إرسال البيانات إلى السيرفر =====
  sendDataToServer(temperature, humidity, gasResistance, irValue);

  Serial.println(F("-----------------------"));
  delay(5000);
}
